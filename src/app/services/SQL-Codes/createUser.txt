CREATE TABLE user (
    id SERIAL PRIMARY KEY,
    username VARCHAR(255),
    password VARCHAR(255),
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    streetName VARCHAR(255),
    houseNumber VARCHAR(10),
    postalCode VARCHAR(10),
    city VARCHAR(255)
);
